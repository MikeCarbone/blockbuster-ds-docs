# Wrapper
Wraps content in a max-width container. Centers upon expanding. This image won't go wider than the specified max-width.

import { Image, Wrapper } from 'blockbuster-design-system'

<Wrapper>
    <Image height="300px" width="100%" src="https://source.unsplash.com/random" alt="Random unsplash pic" />
</Wrapper>


```js
import { Copy } from 'blockbuster-design-system`
    
export default function Home => (
    <Wrapper>
        <Image height="300px" width="100%" src="https://source.unsplash.com/random" alt="Random unsplash pic" />
    </Wrapper>
);
```
